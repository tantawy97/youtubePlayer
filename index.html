overlay.addEventListener('touchend', (e) => { isTouchDevice = true; e.preventDefault(); const currentTime = new Date().getTime(); const tapX = e.changedTouches[0].clientX; const width = overlay.clientWidth; const tapInterval = currentTime - lastTapTime; const controlsHidden = controlsDiv.classList.contains('hidden'); // Double tap detection if (tapInterval < 300 && Math.abs(tapX - lastTapX) < 50) { // Double tap detected -> seek if (tapX > width / 2) player.seekTo(player.getCurrentTime() + 10, true); else player.seekTo(Math.max(player.getCurrentTime() - 10, 0), true); // Prevent pause/play lastTapTime = 0; return; } // Single tap if (controlsHidden) { showControls(); // just show controls } else { // Toggle play/pause if (isPlaying) player.pauseVideo(); else player.playVideo(); } lastTapTime = currentTime; lastTapX = tapX; });
